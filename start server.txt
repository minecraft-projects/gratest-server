-- commands to do to set up server

sudo apt update
sudo apt upgrade
sudo apt install openjdk-21-jdk

mkdir minecraft_server
cd minecraft_server
wget "https://fill-data.papermc.io/v1/objects/8de7c52c3b02403503d16fac58003f1efef7dd7a0256786843927fa92ee57f1e/paper-1.21.8-60.jar"




-- starting server after set up
run this command to start server
cd minecraft_server
java -Xmx12288M -Xms12288M -jar paper.jar nogui
